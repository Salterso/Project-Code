# Create violin plots for all the original metric columns (excluding diff columns)
metric_cols = ["prompt_complexity_score", "burstiness_index", "syntactic_variability", 
               "semantic_coherence_score", "lexical_diversity_ratio", "readability_grade_level", 
               "generation_confidence_score"]

fig, axes = plt.subplots(len(metric_cols), 1, figsize=(10, 3*len(metric_cols)))

for idx, col in enumerate(metric_cols):
    sns.violinplot(x="author_type", y=col, data=df, ax=axes[idx])
    axes[idx].set_title(f"Violin Plot: {col}")

plt.tight_layout()
plt.show()

# Create a correlation matrix between metrics and is_ai column
correlation_with_ai = df[metric_cols + ['is_ai']].corr()['is_ai'].drop('is_ai')

# Create a heatmap
plt.figure(figsize=(10, 6))
sns.heatmap(correlation_with_ai.values.reshape(-1, 1), 
            annot=True, 
            fmt='.3f', 
            cmap='coolwarm', 
            center=0,
            xticklabels=['Correlation with is_ai'],
            yticklabels=metric_cols,
            cbar_kws={'label': 'Correlation Coefficient'})
plt.title('Correlation of Metrics with AI vs Human Classification')
plt.tight_layout()
plt.show()
